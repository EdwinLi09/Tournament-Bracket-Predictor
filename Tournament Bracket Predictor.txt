// SETUP TEAMS SCREEN
function SetupScreen(props) {
  const tournament = props.route.params.tournament;
  const TEAMS_KEY = tournament.teamsKey;
  const BRACKET_KEY = tournament.bracketKey;

  const [teamName, setTeamName] = useState('');
  const [imageUrl, setImageUrl] = useState('');
  const [teamsList, setTeamsList] = useState([]);

  useEffect(() => {
    loadTeams();
  }, []);

  const loadTeams = async () => {
    try {
      const saved = await AsyncStorage.getItem(TEAMS_KEY);
      if (saved !== null) {
        setTeamsList(JSON.parse(saved));
      } else {
        setTeamsList([]);
      }
    } catch (e) {
      setTeamsList([]);
    }
  };

  const addTeam = async () => {
    const trimmedName = teamName.trim();
    const trimmedUrl = imageUrl.trim();

    if (trimmedName === '' || trimmedUrl === '') {
      Alert.alert('Invalid Input', 'Please enter both a team name and an image URL', [{ text: 'OK' }]);
      return;
    }

    if (teamsList.length >= 16) {
      Alert.alert('Limit Reached', 'You can only have 16 teams in the bracket', [{ text: 'OK' }]);
      return;
    }

    try {
      const newTeams = [...teamsList, { id: Math.random().toString(), name: trimmedName, logo: trimmedUrl }];
      await AsyncStorage.setItem(TEAMS_KEY, JSON.stringify(newTeams));
      // Clear bracket since teams changed
      await AsyncStorage.removeItem(BRACKET_KEY);
      setTeamsList(newTeams);
      setTeamName('');
      setImageUrl('');
      Keyboard.dismiss();
    } catch (e) {
      // ignore
    }
  };

  const removeTeam = async (teamID) => {
    try {
      const updated = teamsList.filter((t) => t.id !== teamID);
      await AsyncStorage.setItem(TEAMS_KEY, JSON.stringify(updated));
      // Clear bracket since teams changed
      await AsyncStorage.removeItem(BRACKET_KEY);
      setTeamsList(updated);
    } catch (e) {
      // ignore
    }
  };

  const clearAll = () => {
    Alert.alert('Clear All Teams', 'Are you sure? This will also reset your bracket.', [
      { text: 'Cancel' },
      {
        text: 'Clear',
        onPress: async () => {
          await AsyncStorage.removeItem(TEAMS_KEY);
          await AsyncStorage.removeItem(BRACKET_KEY);
          setTeamsList([]);
        }
      }
    ]);
  };

  return (
    <TouchableWithoutFeedback onPress={() => Keyboard.dismiss()}>
      <View style={styles.setupContainer}>
        <Text style={styles.setupTitle}>{tournament.title}: {teamsList.length} / 16 Teams</Text>
        <Text style={styles.setupHint}>Teams 1-2 play Match 1, teams 3-4 play Match 2, etc.</Text>

        <View style={styles.inputSection}>
          <Text style={styles.inputLabel}>Team Name:</Text>
          <TextInput
            style={styles.textInput}
            value={teamName}
            onChangeText={setTeamName}
            placeholder='e.g. Bayern Munich'
          />
          <Text style={styles.inputLabel}>Image URL:</Text>
          <TextInput
            style={styles.textInput}
            value={imageUrl}
            onChangeText={setImageUrl}
            placeholder='https://...'
          />
          <TouchableOpacity style={styles.addButton} onPress={addTeam}>
            <Text style={styles.addButtonText}>Add Team</Text>
          </TouchableOpacity>
        </View>

        <ScrollView style={styles.teamsScroll} keyboardShouldPersistTaps='handled'>
          {teamsList.map((team, idx) => (
            <TouchableOpacity key={team.id} style={styles.teamRow} onPress={() => removeTeam(team.id)}>
              <Text style={styles.teamRowNum}>{idx + 1}.</Text>
              <Image style={styles.teamRowLogo} source={{ uri: team.logo }} />
              <Text style={styles.teamRowName}>{team.name}</Text>
              <Text style={styles.teamRowDelete}>tap to remove</Text>
            </TouchableOpacity>
          ))}
        </ScrollView>

        {teamsList.length > 0 && (
          <TouchableOpacity style={styles.clearAllButton} onPress={clearAll}>
            <Text style={styles.clearAllText}>Clear All Teams</Text>
          </TouchableOpacity>
        )}
      </View>
    </TouchableWithoutFeedback>
  );
}

// MATCH BOX COMPONENT 
function MatchBox(props) {
  const team1 = props.team1;
  const team2 = props.team2;
  const winner = props.winner;

  const isTeam1Winner = winner && team1 && winner.id === team1.id;
  const isTeam2Winner = winner && team2 && winner.id === team2.id;
  const canSelect = team1 && team2;

  return (
    <View style={styles.matchBox}>
      <Text style={styles.matchLabel}>{props.matchLabel}</Text>
      <View style={styles.matchRow}>
        {team1 && (
          <TouchableOpacity
            style={[styles.teamCard, isTeam1Winner && styles.selectedTeam]}
            onPress={() => { if (canSelect) props.onSelect(team1); }}
          >
            <Image style={styles.teamLogo} source={{ uri: team1.logo }} />
            <Text style={styles.teamName}>{team1.name}</Text>
          </TouchableOpacity>
        )}
        {!team1 && (
          <View style={[styles.teamCard, styles.tbdTeam]}>
            <Text style={styles.tbdText}>TBD</Text>
          </View>
        )}

        <Text style={styles.vsText}>vs</Text>

        {team2 && (
          <TouchableOpacity
            style={[styles.teamCard, isTeam2Winner && styles.selectedTeam]}
            onPress={() => { if (canSelect) props.onSelect(team2); }}
          >
            <Image style={styles.teamLogo} source={{ uri: team2.logo }} />
            <Text style={styles.teamName}>{team2.name}</Text>
          </TouchableOpacity>
        )}
        {!team2 && (
          <View style={[styles.teamCard, styles.tbdTeam]}>
            <Text style={styles.tbdText}>TBD</Text>
          </View>
        )}
      </View>
    </View>
  );
}

//  BRACKET SCREEN
function BracketScreen(props) {
  const tournament = props.route.params.tournament;
  const TEAMS_KEY = tournament.teamsKey;
  const BRACKET_KEY = tournament.bracketKey;

  const [teams, setTeams] = useState([]);
  const [r16Winners, setR16Winners] = useState([null, null, null, null, null, null, null, null]);
  const [qfWinners, setQfWinners] = useState([null, null, null, null]);
  const [sfWinners, setSfWinners] = useState([null, null]);
  const [champion, setChampion] = useState(null);

  useEffect(() => {
    loadTeams();
    loadBracket();
  }, []);

  const loadTeams = async () => {
    try {
      const saved = await AsyncStorage.getItem(TEAMS_KEY);
      if (saved !== null) {
        setTeams(JSON.parse(saved));
      }
    } catch (e) {
      // ignore
    }
  };

  const loadBracket = async () => {
    try {
      const saved = await AsyncStorage.getItem(BRACKET_KEY);
      if (saved !== null) {
        const data = JSON.parse(saved);
        setR16Winners(data.r16);
        setQfWinners(data.qf);
        setSfWinners(data.sf);
        setChampion(data.champion);
      }
    } catch (e) {
      // ignore
    }
  };

  const saveBracket = async (r16, qf, sf, champ) => {
    try {
      const data = { r16: r16, qf: qf, sf: sf, champion: champ };
      await AsyncStorage.setItem(BRACKET_KEY, JSON.stringify(data));
    } catch (e) {
      // ignore
    }
  };

  // Selecting a Round of 16 winner. If changing an existing pick,
  // cascade-reset the dependent Quarter Final, Semi Final, and Champion.
  const selectR16Winner = (matchIdx, team) => {
    const newR16 = [...r16Winners];
    const oldWinner = newR16[matchIdx];
    newR16[matchIdx] = team;

    let newQf = [...qfWinners];
    let newSf = [...sfWinners];
    let newChamp = champion;

    if (oldWinner && oldWinner.id !== team.id) {
      const qfIdx = Math.floor(matchIdx / 2);
      newQf[qfIdx] = null;
      const sfIdx = Math.floor(qfIdx / 2);
      newSf[sfIdx] = null;
      newChamp = null;
    }

    setR16Winners(newR16);
    setQfWinners(newQf);
    setSfWinners(newSf);
    setChampion(newChamp);
    saveBracket(newR16, newQf, newSf, newChamp);
  };

  const selectQfWinner = (matchIdx, team) => {
    const newQf = [...qfWinners];
    const oldWinner = newQf[matchIdx];
    newQf[matchIdx] = team;

    let newSf = [...sfWinners];
    let newChamp = champion;

    if (oldWinner && oldWinner.id !== team.id) {
      const sfIdx = Math.floor(matchIdx / 2);
      newSf[sfIdx] = null;
      newChamp = null;
    }

    setQfWinners(newQf);
    setSfWinners(newSf);
    setChampion(newChamp);
    saveBracket(r16Winners, newQf, newSf, newChamp);
  };

  const selectSfWinner = (matchIdx, team) => {
    const newSf = [...sfWinners];
    const oldWinner = newSf[matchIdx];
    newSf[matchIdx] = team;

    let newChamp = champion;
    if (oldWinner && oldWinner.id !== team.id) {
      newChamp = null;
    }

    setSfWinners(newSf);
    setChampion(newChamp);
    saveBracket(r16Winners, qfWinners, newSf, newChamp);
  };

  const selectChampion = (team) => {
    setChampion(team);
    saveBracket(r16Winners, qfWinners, sfWinners, team);
  };

  const resetBracket = () => {
    Alert.alert('Reset Bracket', 'Clear all predictions?', [
      { text: 'Cancel' },
      {
        text: 'Reset',
        onPress: async () => {
          const emptyR16 = [null, null, null, null, null, null, null, null];
          const emptyQf = [null, null, null, null];
          const emptySf = [null, null];
          setR16Winners(emptyR16);
          setQfWinners(emptyQf);
          setSfWinners(emptySf);
          setChampion(null);
          await AsyncStorage.removeItem(BRACKET_KEY);
        }
      }
    ]);
  };

  // If user has not added 16 teams yet, show message
  if (teams.length < 16) {
    return (
      <View style={styles.emptyContainer}>
        <Text style={styles.emptyTitle}>Not Enough Teams</Text>
        <Text style={styles.emptyText}>You have {teams.length}/16 teams.{'\n'}Please add {16 - teams.length} more.</Text>
        <TouchableOpacity style={[styles.homeButton, { borderColor: 'blue' }]} onPress={() => props.navigation.navigate('Setup', { tournament: tournament })}>
          <Text style={[styles.homeButtonText, { color: 'blue' }]}>Go to Setup</Text>
        </TouchableOpacity>
      </View>
    );
  }

  // Compute later rounds from earlier winners
  const qfMatches = [
    { team1: r16Winners[0], team2: r16Winners[1] },
    { team1: r16Winners[2], team2: r16Winners[3] },
    { team1: r16Winners[4], team2: r16Winners[5] },
    { team1: r16Winners[6], team2: r16Winners[7] }
  ];

  const sfMatches = [
    { team1: qfWinners[0], team2: qfWinners[1] },
    { team1: qfWinners[2], team2: qfWinners[3] }
  ];

  const finalMatch = { team1: sfWinners[0], team2: sfWinners[1] };

  const r16Count = r16Winners.filter((x) => x !== null).length;
  const qfCount = qfWinners.filter((x) => x !== null).length;
  const sfCount = sfWinners.filter((x) => x !== null).length;
  let champCount = 0;
  if (champion) {
    champCount = 1;
  }
  const totalPicks = r16Count + qfCount + sfCount + champCount;

  return (
    <ScrollView style={styles.bracketScroll} contentContainerStyle={{ paddingBottom: 40 }}>
      <View style={styles.progressBar}>
        <Text style={styles.progressText}>{tournament.title} Picks: {totalPicks} / 15</Text>
      </View>

      <Text style={styles.roundTitle}>Round of 16</Text>
      {round16Matchups.map((pair, idx) => (
        <MatchBox
          key={'r16-' + idx}
          team1={teams[pair[0]]}
          team2={teams[pair[1]]}
          winner={r16Winners[idx]}
          matchLabel={'Match ' + (idx + 1)}
          onSelect={(team) => selectR16Winner(idx, team)}
        />
      ))}

      <Text style={styles.roundTitle}>Quarter Finals</Text>
      {qfMatches.map((match, idx) => (
        <MatchBox
          key={'qf-' + idx}
          team1={match.team1}
          team2={match.team2}
          winner={qfWinners[idx]}
          matchLabel={'QF ' + (idx + 1)}
          onSelect={(team) => selectQfWinner(idx, team)}
        />
      ))}

      <Text style={styles.roundTitle}>Semi Finals</Text>
      {sfMatches.map((match, idx) => (
        <MatchBox
          key={'sf-' + idx}
          team1={match.team1}
          team2={match.team2}
          winner={sfWinners[idx]}
          matchLabel={'SF ' + (idx + 1)}
          onSelect={(team) => selectSfWinner(idx, team)}
        />
      ))}

      <Text style={styles.roundTitle}>Final</Text>
      <MatchBox
        team1={finalMatch.team1}
        team2={finalMatch.team2}
        winner={champion}
        matchLabel={tournament.title + ' Final'}
        onSelect={(team) => selectChampion(team)}
      />

      {champion && (
        <View style={styles.championBox}>
          <Image style={styles.trophyImageSmall} source={{ uri: tournament.logo }} />
          <Text style={styles.championTitle}>Your Champion</Text>
          <Image style={styles.championLogo} source={{ uri: champion.logo }} />
          <Text style={styles.championName}>{champion.name}</Text>
        </View>
      )}

      <TouchableOpacity style={styles.resetButton} onPress={resetBracket}>
        <Text style={styles.resetText}>Reset Bracket</Text>
      </TouchableOpacity>
    </ScrollView>
  );
}

// SUMMARY SCREEN 
function SummaryScreen(props) {
  const tournament = props.route.params.tournament;
  const BRACKET_KEY = tournament.bracketKey;

  const [data, setData] = useState(null);

  useEffect(() => {
    loadData();
  }, []);

  const loadData = async () => {
    try {
      const saved = await AsyncStorage.getItem(BRACKET_KEY);
      if (saved !== null) {
        setData(JSON.parse(saved));
      } else {
        setData(null);
      }
    } catch (e) {
      setData(null);
    }
  };

  const clearPredictions = () => {
    Alert.alert('Clear Predictions', 'Are you sure you want to clear all your predictions?', [
      { text: 'Cancel' },
      {
        text: 'Clear',
        onPress: async () => {
          await AsyncStorage.removeItem(BRACKET_KEY);
          setData(null);
        }
      }
    ]);
  };

  if (data === null) {
    return (
      <View style={styles.emptyContainer}>
        <Text style={styles.emptyTitle}>No Predictions Yet</Text>
        <Text style={styles.emptyText}>Go fill out your {tournament.title} bracket first!</Text>
      </View>
    );
  }

  return (
    <ScrollView style={styles.summaryScroll} contentContainerStyle={{ padding: 20 }}>
      {data.champion && (
        <View style={styles.summaryChampion}>
          <Image style={styles.trophyImageSmall} source={{ uri: tournament.logo }} />
          <Text style={styles.summarySectionTitle}>Your Predicted {tournament.title} Champion</Text>
          <Image style={styles.summaryChampLogo} source={{ uri: data.champion.logo }} />
          <Text style={styles.summaryChampName}>{data.champion.name}</Text>
        </View>
      )}

      <Text style={styles.summarySection}>Round of 16 Winners</Text>
      {data.r16.map((team, idx) => (
        <View key={'sr16-' + idx} style={styles.summaryRow}>
          <Text style={styles.summaryMatchLabel}>Match {idx + 1}:</Text>
          {team && (
            <View style={styles.summaryTeam}>
              <Image style={styles.summaryLogo} source={{ uri: team.logo }} />
              <Text style={styles.summaryTeamName}>{team.name}</Text>
            </View>
          )}
          {!team && (
            <Text style={styles.summaryUnpicked}>Not picked</Text>
          )}
        </View>
      ))}

      <Text style={styles.summarySection}>Quarter Finals Winners</Text>
      {data.qf.map((team, idx) => (
        <View key={'sqf-' + idx} style={styles.summaryRow}>
          <Text style={styles.summaryMatchLabel}>QF {idx + 1}:</Text>
          {team && (
            <View style={styles.summaryTeam}>
              <Image style={styles.summaryLogo} source={{ uri: team.logo }} />
              <Text style={styles.summaryTeamName}>{team.name}</Text>
            </View>
          )}
          {!team && (
            <Text style={styles.summaryUnpicked}>Not picked</Text>
          )}
        </View>
      ))}

      <Text style={styles.summarySection}>Semi Finals Winners</Text>
      {data.sf.map((team, idx) => (
        <View key={'ssf-' + idx} style={styles.summaryRow}>
          <Text style={styles.summaryMatchLabel}>SF {idx + 1}:</Text>
          {team && (
            <View style={styles.summaryTeam}>
              <Image style={styles.summaryLogo} source={{ uri: team.logo }} />
              <Text style={styles.summaryTeamName}>{team.name}</Text>
            </View>
          )}
          {!team && (
            <Text style={styles.summaryUnpicked}>Not picked</Text>
          )}
        </View>
      ))}

      <TouchableOpacity style={styles.clearPredictionsButton} onPress={clearPredictions}>
        <Text style={styles.clearPredictionsText}>Clear Predictions</Text>
      </TouchableOpacity>
    </ScrollView>
  );
}
